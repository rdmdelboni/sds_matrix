37def _default_data_root() -> Path: 

38 """Choose a user-writable data directory. 

39 

40 Priority: 

41 1) FDS_DATA_DIR env var, if set 

42 2) On Windows: %APPDATA%/FDS Reader 

43 Else: ~/.local/share/fds_reader 

44 3) Fallback to repo-local ./data (useful during development) 

45 """ 

46 # 1) explicit override 

47 override = os.getenv("FDS_DATA_DIR") 

48 if override: 

49 return Path(override) 

50 

51 # Detect frozen builds or site-packages install 

52 is_frozen = bool(getattr(sys, "frozen", False)) 

53 base_str = str(BASE_DIR).lower() 

54 in_site_packages = any( 

55 name in base_str for name in ("site-packages", "dist-packages") 

56 ) 

57 base_writable = os.access(BASE_DIR, os.W_OK) 

58 

59 # 2) prefer user data dir when frozen/installed or base is not writable 

60 if is_frozen or in_site_packages or not base_writable: 

61 if os.name == "nt": 

62 appdata = os.getenv("APPDATA") 

63 if appdata: 

64 return Path(appdata) / "FDS Reader" 

65 # Fallback for Windows without APPDATA 

66 return Path.home() / "AppData" / "Roaming" / "FDS Reader" 

67 # POSIX fallback 

68 return Path.home() / ".local" / "share" / "fds_reader" 

69 

70 # 3) dev default 

71 return BASE_DIR / "data" 

72
